\frame{\frametitle{Estilos de numeração de páginas}
\begin{itemizespc}
\item Comando:
\begin{itemize}
\item $\backslash pagestyle\{ estilo \}$
\item $\backslash thispagestyle\{ estilo \}$
\end{itemize}
\item estilos disponíveis:
\begin{itemize}
\item {\tt plain}: número de página no rodapé é centralizado
\item {\tt headings}: capítulo corrente e número da página no cabeçalho
\item {\tt empty}: cabeçalho e rodapé vazios (página ``limpa'');
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Alinhamento do texto}
\begin{itemizespc}
\item Comando:
\begin{itemize}
\item $\backslash \mathtt{begin}\{alinhamento\}$ \\
      ... \\
      $ \backslash \mathtt{end} \{alinhamento\}$
\end{itemize}
\item em que $alinhamento$ pode ser:
\begin{itemize}
\item {\tt flushleft}: alinhado à margem esquerda
\item {\tt flushright}: alinhado à margem direita
\item {\tt center}: centralizado
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Símbolos especiais e traços}
\begin{itemizespc}
\item Para utilizar alguns símbolos especiais, deve-se utilizar o $\backslash$ antes de cada símbolo:
\begin{center}
$\textbackslash$\^ \/ \# \$ \%  \_ \{ \}
\end{center}
\item Deve-se inserir ``$\backslash$'' antes de cada símbolo no decorrer do texto:
\begin{itemize}
\item ``$\backslash\#$'' para mostrar \#
\end{itemize}
\item Traços:
\begin{itemize}
\item ``- '' aparecerá como -
\item ``- - '' aparecerá como -- 
\item ``- - - '' aparecerá como ---
\end{itemize}
\item Escrever sentença entre aspas: $\grave{ }$ $\grave{ }$ {\it texto} ' ' 
\begin{itemize}
\item $\grave{ }$ $\grave{ }$ A vida é bela.' ' produz ``A vida é bela.''
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Cores}
\begin{itemizespc}
\item Para utilizar cores no texto do Latex, é necessário utilizar o pacote {\tt color}:
\begin{itemize}
\item $\backslash \mathtt{usepackage}\{ \mathtt{color}\}$
\end{itemize}
\item Definir a cor no preâmbulo do documento Latex de acordo com o modelo RGB, isto é, valores entre $[0,1]$ para cada um dos canais (vermelho - red, verde - green e azul - blue)
\begin{itemize}
\item $\backslash \mathtt{definecolor}\{ \textrm{nome da cor}\}\{ \mathtt{rgb}\}\{ \mathtt{0.5,0.25,0.1}\}$
\item vermelho $= 0.5$
\item verde $= 0.25$
\item azul $= 0.1$
\item Lembre-se: sempre utilizar ponto ao invés de vírgula!!!
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Cores}
\begin{itemizespc}
\item Para colorir uma frase ou palavra no texto, utilizar o comando abaixo no texto:
\begin{itemize}
\item $\backslash \mathtt{textcolor}\{ \textrm{nome da cor}\}\{ \mathtt{texto}\}$
\end{itemize}
\item Exemplo:
\begin{itemize}
\item No preâmbulo: $\backslash \mathtt{definecolor}\{ \textrm{qualquer}\}\{ \mathtt{rgb}\}\{ \mathtt{0.5,0.25,0.1}\}$
\item No texto: $\backslash \mathtt{textcolor}\{ \textrm{qualquer}\} \{ \textrm{ Aeeee, colorido!!!}\}$
\item Resultado: \textcolor{qualquer}{Aeeee, colorido!!!}
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Espaços e medidas}
\begin{itemizespc}
\item Para adicionar espaço ``vazio'' entre as palavras, frases e figuras, utilize o comando:\\
\begin{center}
$\backslash \mathtt{hspace}\{ medida\}$
\end{center}
\begin{itemize}
\item $medida$ pode ser em $cm$, $in$ ou $pt$
\end{itemize}
\item Exemplo:
\begin{itemize}
\item Aqui temos $\backslash \mathtt{hspace}\{ \mathtt{2cm}\}$ de espaço
\item Resultado: Aqui temos \hspace{2cm} de espaço.
\end{itemize}
\item Pode ser combinado com o comando $\backslash \mathtt{underline}\{ \mathtt{2cm}\}$
\begin{itemize}
\item Aqui temos $\backslash \mathtt{hspace}\{ \mathtt{hspace}\{ \mathtt{2cm}\}\}$ de espaço sublinhado
\item Resultado: Aqui temos \underline{\hspace{2cm}} de espaço sublinhado.
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Espaços e medidas}
\begin{itemizespc}
\item Para adicionar espaço ``vazio'' entre as linhas, isto é, na vertical:\\
\begin{center}
$\backslash \mathtt{vspace}\{ medida\}$
\end{center}
\item Exemplo:
\begin{itemize}
\item Aqui temos $\backslash \backslash \quad \backslash \mathtt{vspace}\{ \mathtt{2cm}\}$ de espaço
\item Resultado: Aqui temos\\ \vspace{1cm} de espaço.
\end{itemize}
\end{itemizespc}
}


\frame{\frametitle{Imagens: o comando includegraphics}

\frame{\frametitle{Imagens: o comando figure}
\begin{itemizespc}
\item Para trabalhar com figuras \LaTeX\, utiliza-se o ambiente {\texttt figure}:\\
\begin{flushleft}
$\backslash \mathtt{begin}\{\mathtt{figure}\}[pos] $\\
$\backslash \mathtt{includegraphics}[medidas]\{ \textrm{nome do arquivo} \}$
$\backslash \mathtt{caption} \{ \textrm{Legenda da figura} \}$ \\
$\backslash \mathtt{label}\{\textrm{rotulo}\}$ \\
$\backslash \mathtt{end}\{\mathtt{figure}\}$
\end{flushleft}
\item em que $pos$ pode ser:
\begin{itemize}
\item b - bottom (tabela ficará na parte de baixo da página)
\item t - top (tabela ficará na parte de cima da página)
\item h - here (tabela ficará onde digitada na página)
\item p - (tabela ficará em parte separada)
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Imagens: o comando figure}
\begin{itemizespc}
\item Para trabalhar com figuras \LaTeX\, utiliza-se o ambiente {\texttt figure}:\\
\begin{flushleft}
$\backslash \mathtt{begin}\{\mathtt{figure}\}[pos] $\\
$\backslash \mathtt{includegraphics}[medidas]\{ \textrm{nome do arquivo} \}$
$\backslash \mathtt{caption} \{ \textrm{Legenda da figura} \}$ \\
$\backslash \mathtt{label}\{\textrm{rotulo}\}$ \\
$\backslash \mathtt{end}\{\mathtt{figure}\}$
\end{flushleft}
\item Alguns exemplos para $medidas$:
\begin{itemize}
\item $[width=x\backslash \mathtt{linewidth}]$: ajuste de acordo com a largura da linha
\item $[scale=x]$: escala da imagem
\item $x$ é um valor maior do que zero. Se $x < 1$, a imagem é subdimensionada, enquanto que, $x > 1$, a imagem é aumentada.
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Ambiente matemático: fórmulas}

\$ $<$ fórmula matemática $>$ \$%$\backslash\mathtt{footnote}\{\textrm{ texto }\}$
\end{center}
\item Exemplo:\\

{\scriptsize 
\begin{columns}[T]

\end{column}
\begin{column}{.45\textwidth}

\end{itemizespc}
}


\frame{\frametitle{Ambiente matemático: subescrito e sobrescritos}
\begin{itemizespc}
\item Subescrito é escrito como:
\begin{itemize}
\item \$ 4\_{}$\{$2$\}$\$ produz $4_2$
\end{itemize}
\item Sobrescrito é escrito como:
\begin{itemize}
\item \$ 4\^{}$\{$2$\}$\$ produz $4^2$
\end{itemize}
\item Observação: quando o expoente possuir um único digito, 4\^{}$\{$2$\}$ (ou 4\_{}$\{$2$\}$) pode ser reescrito como 4\^{}2 (ou 4\_2)
\end{itemizespc}
}

\frame{\frametitle{Fórmulas matemáticas: ambiente array}
\begin{itemizespc}
\item Ambiente útil para escrever matrizes e condicionais, pois ocorre a organização dos elementos em linhas e colunas
\begin{center}
$\backslash \mathtt{begin}\{\mathtt{array}\}\{cc\}$ \\
$\quad$ $\quad a \quad \& \quad b \backslash \backslash$ \\

%$\backslash \mathtt{label}\{\textrm{rotulo}\}$ \\

\end{center}
\item produz
\begin{equation}
\begin{array}{cc}
