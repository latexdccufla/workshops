\frame{\frametitle{Modularização de documentos}
\begin{itemizespc}
\item Documentos muito grandes podem ser difíceis de manusear
\item {\bf Solução:} dividir o arquivo ``principal'' \LaTeX\ em arquivos menores
\item Ideia é semelhante comparada à modularização na linguagem C por exemplo
\begin{itemize}
\item Dividir o conteúdo do arquivo .tex principal e incluí-los em arquivos .tex
\item Utilizar o comando $\backslash {\tt include} \{ \textrm{nome do arquivo sem o .tex} \}$ para incluir cada arquivo .tex
\item Cada arquivo .tex é incluído após $\backslash {\tt begin}\{ {\tt document }\}$
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Modularização de documentos}
\begin{itemizespc}
\item Cada arquivo .tex gerado a ser incluído no arquivo \LaTeX\ principal:
\begin{itemize}
\item Não pode conter os comandos que aparecem no preâmbulo
\item Não podem conter $\backslash {\tt begin}\{ {\tt document }\}$ e $\backslash {\tt end}\{ {\tt document }\}$
\end{itemize}
\item {\bf Sugestão:} separar cada capítulo (ou seção) do documento em um arquivo .tex
\item Variações em relação ao $\backslash {\tt include}$
\begin{itemize}
\item $\backslash {\tt input}$: não inicia o conteúdo do arquivo em uma nova página
\item $\backslash {\tt includeonly}\{ arquivo1,arquivo2,...,arquivo_N\}$
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Bibliografia (BibTeX)}
\begin{itemizespc}
\item BibTeX é uma ferramenta de formatação de bibliografias usada em documentos LaTeX
\item Ideia básica:
\begin{itemize}
\item Criar um arquivo de bibliografias com formato .bib 
\item Inserir as referências bibliográficas formatadas
\item Incluir o comando $\backslash \mathtt{bibliographystyle}\{estilo\}$ no preâmbulo do documento
\item Inserir o comando $\backslash \mathtt{bibliography}\{ \textrm{arquivo sem .bib} \}$ no local adequado dentro de $\backslash \mathtt{begin}\{ \textrm{document} \}$
\item Compilar o arquivo principal do Latex e em seguida compilar o arquivo .bib
\item Recompilar o arquivo principal do Latex
\end{itemize}
\end{itemizespc}
}

\frame{\frametitle{Bibliografia (BibTeX)}
\begin{itemizespc}
\item Formato geral de uma referência do tipo {\tt inproceedings}:
\begin{flushleft}
   $@$ {\tt inproceedings}\{identificador da referência,\\
   $\quad$ {\tt title}=\{ título do trabalho \},\\
   $\quad$ {\tt author}=\{ autores (cada um separado por {\it and}) \},\\
   $\quad$ {\tt booktitle}=\{ nome da conferência \},\\
   $\quad$ {\tt pages}=\{ página inicial - - página final \},\\
   $\quad$ {\tt year}=\{ ano \} \\
\}
\end{flushleft}
\item Existem outros campos opcionais, como {\tt organization} etc.
\end{itemizespc}
}

\frame{\frametitle{Bibliografia (BibTeX)}
\begin{itemizespc}
\item Formato geral de uma referência do tipo {\tt article}:
\begin{flushleft}
   $@$ {\tt article}\{identificador da referência,\\
   $\quad$ {\tt title}=\{ título do trabalho \},\\
   $\quad$ {\tt author}=\{ autores (cada um separado por {\it and}) \},\\
   $\quad$ {\tt journal}=\{ nome do periódico \},\\
   $\quad$ {\tt volume}=\{ volume \},\\
   $\quad$ {\tt pages}=\{ página inicial - - página final \},\\
   $\quad$ {\tt year}=\{ ano \} \\
\}
\end{flushleft}
\item Existem outros campos opcionais, como {\tt number}, {\tt publisher} etc.
\end{itemizespc}
}

\frame{\frametitle{Bibliografia (BibTeX)}
\begin{itemizespc}
\item Outros tipos
\begin{itemize}
\item {\tt inbook}: parte de um livro
\item {\tt incollection}: parte de um livro que possui título próprio
\item {\tt manual}: documento técnico
\item {\tt mastersthesis}: dissertação de mestrado

\frame{\frametitle{Algoritmos}
\begin{itemizespc}
\item É possível incluir blocos que descrevem algoritmos em pseudo-código em um documento \LaTeX
\item Existem vários pacotes e o {\tt algorithm2e} é bastante popular
\begin{itemize}
\item Incluir o pacote no preâmbulo: $\backslash \mathtt{usepackage}[]\backslash\{\mathtt{algorithm2e}\}$
\end{itemize}
\item O ambiente é denominado $\backslash \mathtt{begin}\{ \mathtt{algorithm} \}$ ... $\backslash \mathtt{end}\{ \mathtt{algorithm} \}$
\end{itemizespc}
}
